5) GSAP: Card Stack + Swipe
Goal: stacked idle look + swipe transition between cards (keep your API flow).

 On load, set staggered rotation/offset for back & mid cards; keep top card upright.

 On swipe left/right:

animate card--top off-screen with slight rotation & fade,

promote card--mid → card--top (reset rotation, remove blur, apply gradient),

promote card--back → card--mid (reduce blur),

append a new back card from API and place as card--back with small rotated offset and blur.

js
Copy
Edit
// Setup
gsap.set(".card--back", { rotation:-8, yPercent:-2, filter:"blur(0.6px)", zIndex:1 });
gsap.set(".card--mid",  { rotation: 8, yPercent: 2, filter:"blur(0.3px)", zIndex:2 });
gsap.set(".card--top",  { rotation: 0, yPercent: 0, filter:"none", zIndex:3 });

function swipe(direction = 1) {
  const top = document.querySelector(".card--top");
  const mid = document.querySelector(".card--mid");
  const back = document.querySelector(".card--back");

  // animate top out
  gsap.to(top, {
    x: direction * window.innerWidth,
    y: -30,
    rotation: direction * 10,
    opacity: 0,
    duration: 0.35,
    ease: "power2.in",
    onComplete: () => {
      top.remove(); // or recycle to back with new content
      promote();
    }
  });

  function promote() {
    // mid -> top
    mid.classList.replace("card--mid", "card--top");
    gsap.to(mid, { rotation:0, yPercent:0, filter:"none", duration:0.25, ease:"power2.out" });

    // back -> mid
    back.classList.replace("card--back", "card--mid");
    gsap.to(back, { rotation:8, yPercent:2, filter:"blur(0.3px)", duration:0.25 });

    // inject new back from API
    const next = buildCardFromAPI(getNextCard()); // you already have this
    next.classList.add("card","card--back");
    document.querySelector(".deck").prepend(next);
    gsap.set(next, { rotation:-8, yPercent:-2, filter:"blur(0.6px)", opacity:0 });
    gsap.to(next, { opacity:1, duration:0.2 });
  }
}
 Remove the current “single card swipe” look; ensure three-card presence is always visible.
